Mouse clicking games

August 23rd, 2014

Do you know games like cookie clicker where you only have to click a lot?

Cookie Clicker

You can do that automatically with the following Python script.

Just execute xdotool getmouselocation --shell before that to find the position of your mouse.

 1#!/usr/bin/env python
 2
 3"""Tool to make automatic clicks VERY fast (useful for idle games)."""
 4
 5import os
 6import time
 7import random
 8
 9
10def main(clicks, twiggle, x, y, delay):
11    for i in range(clicks):
12        xrand = random.randint(-twiggle, twiggle)
13        yrand = random.randint(-twiggle, twiggle)
14        xpos, ypos = x+xrand, y+yrand
15        os.system("xdotool mousemove %i %i click 1" % (xpos, ypos))
16        if delay > 0:
17            time.sleep(delay)
18
19
20if __name__ == '__main__':
21    from argparse import ArgumentParser
22
23    parser = ArgumentParser()
24
25    # Add more options if you like
26    parser.add_argument("-c", "--clicks", dest="clicks",
27                        help="number of clicks that will be done",
28                        default=500,
29                        type=int)
30    parser.add_argument("-t", "--twiggle", dest="twiggle",
31                        help="how much should the cursor move randomly",
32                        default=50,
33                        type=int)
34    parser.add_argument("-x", dest="x",
35                        help="x coordinate where to click",
36                        default=711,
37                        type=int)
38    parser.add_argument("-y", dest="y",
39                        help="y coordinate where to click",
40                        default=467,
41                        type=int)
42    parser.add_argument("--delay", dest="delay",
43                        help="delay between clicks",
44                        default=0.01,
45                        type=float)
46    args = parser.parse_args()
47    main(args.clicks, args.twiggle, args.x, args.y, args.delay)
48
