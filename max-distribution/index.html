Distribution of Random Variables when max gets applied

January 22nd, 2015

I just wanted to solve an exercise where I had random variables $X_1, \dots, X_n$ which were all $U([0, 1])$ distributed and $Y_n = \max(X_1, \dots, X_n)$.

I wondered what the distribution of $Y_n$ is (for big $n$), so I wanted to plot it. How do I plot it? With Python, of course ☺

Here is the program:

 1#!/usr/bin/env python
 2
 3import matplotlib.pyplot as plt
 4import numpy.random
 5
 6
 7def main():
 8    # Generate Data
 9    n = 10000
10    numbers_a = numpy.random.uniform(size=samples)
11    numbers_b = numpy.random.uniform(size=samples)
12    numbers_c = numpy.random.uniform(size=samples)
13    numbers_max = [max(a, b, c)
14                   for a, b, c in zip(numbers_a, numbers_b, numbers_c)]
15
16    # Plot data
17    plt.hist(numbers_max)
18    plt.title("Histogram")
19    plt.xlabel("value")
20    plt.ylabel("count")
21    plt.show()
22

24if __name__ == '__main__':

Improved version

I’ve also created an improved version which makes nicer plots, but might be harder to read:

 1#!/usr/bin/env python
 2
 3import matplotlib.pyplot as plt
 4import numpy.random
 5
 6
 7def main(samples, n):
 8    # Generate Data
 9    numbers_l = []
10    n = n + 1
11    for _ in range(n):
12        numbers_l.append(numpy.random.uniform(size=samples))
13
14    # Plot data
15    plots = []
16    for i in range(1, n):
17        # Build data structure
18        sublist = numbers_l[:i]
19        max_list = numbers_l[0]
20        for numbers_list in sublist:
21            for i, el in enumerate(numbers_list):
22                max_list[i] = max(max_list[i], el)
23        plot_i = plt.hist(max_list, histtype='step', bins=200)
24        plots.append(plot_i)
25    plt.title("Histogram")
26    plt.xlabel("value")
27    plt.ylabel("count")
28    plt.show()
29
30
31def get_parser():
32    from argparse import ArgumentParser, ArgumentDefaultsHelpFormatter
33    parser = ArgumentParser(description=__doc__,
34                            formatter_class=ArgumentDefaultsHelpFormatter)
35    parser.add_argument("-n",
36                        dest="n",
37                        default=3,
38                        type=int,
39                        help="n")
40    parser.add_argument("-s", "--samples",
41                        dest="samples",
42                        default=1000,
43                        type=int,
44                        help="number of samples per Y_n")
45    return parser
46
47
48if __name__ == '__main__':
49    args = get_parser().parse_args()
50    main(args.samples, args.n)

When you call this script with

$ ./plot-random-variable.py -n 4 --samples 100000

it gives

Plot of the maximum of 1 - 4 uniformly distributed variables with 10000 samples
