Classification example

The following is a slightly modified example from the documentation. It shows
how PyBrain starts learning to classify 2-dimensional datapoints into 3 classes:

  1#!/usr/bin/env python
  2# -*- coding: utf-8 -*-
  3
  4# Source: http://pybrain.org/docs/tutorial/fnn.html
  5from pybrain.datasets import ClassificationDataSet
  6from pybrain.utilities import percentError
  7from pybrain.tools.shortcuts import buildNetwork
  8from pybrain.supervised.trainers import BackpropTrainer
  9from pybrain.structure.modules import SoftmaxLayer
 10
 11# Only needed for data generation and graphical output
 12from pylab import ion, ioff, figure, draw, contourf, clf, show, plot
 13from scipy import diag, arange, meshgrid, where
 14from numpy.random import multivariate_normal
 15from random import normalvariate
 16
 17
 18def generate_data(n=400):
 19    INPUT_FEATURES = 2
 20    CLASSES = 3
 21    means = [(-1, 0), (2, 4), (3, 1)]
 22    cov = [diag([1, 1]), diag([0.5, 1.2]), diag([1.5, 0.7])]
 23    alldata = ClassificationDataSet(INPUT_FEATURES, 1, nb_classes=CLASSES)
 24    minX, maxX = means[0][0], means[0][0]
 25    minY, maxY = means[0][1], means[0][1]
 26    for i in range(n):
 27        for klass in range(CLASSES):
 28            features = multivariate_normal(means[klass], cov[klass])
 29            x, y = features
 30            minX, maxX = min(minX, x), max(maxX, x)
 31            minY, maxY = min(minY, y), max(maxY, y)
 32            alldata.addSample(features, [klass])
 33    return {'minX': minX, 'maxX': maxX,
 34            'minY': minY, 'maxY': maxY, 'd': alldata}
 35
 36
 37def generate_data2(n=400):
 38    alldata = ClassificationDataSet(2, 1, nb_classes=2)
 39    minX, maxX = 3, 3
 40    minY, maxY = 2, 2
 41    for i in range(1000):
 42        x = normalvariate(3, 0.6)
 43        y = normalvariate(2, 1)
 44        minX, maxX = min(minX, x), max(maxX, x)
 45        minY, maxY = min(minY, y), max(maxY, y)
 46        alldata.addSample((x, y), (0,))
 47    for i in range(1000):
 48        x = normalvariate(7, 0.5)
 49        y = normalvariate(1, 0.1)
 50        alldata.addSample((x, y), (1,))
 51    return {'minX': minX, 'maxX': maxX,
 52            'minY': minY, 'maxY': maxY, 'd': alldata}
 53
 54
 55def perceptron(hidden_neurons=5, weightdecay=0.01, momentum=0.1):
 56    INPUT_FEATURES = 2
 57    CLASSES = 3
 58    HIDDEN_NEURONS = hidden_neurons
 59    WEIGHTDECAY = weightdecay
 60    MOMENTUM = momentum
 61
 62    # Generate the labeled set
 63    g = generate_data()
 64    #g = generate_data2()
 65    alldata = g['d']
 66    minX, maxX, minY, maxY = g['minX'], g['maxX'], g['minY'], g['maxY']
 67
 68    # Split data into test and training dataset
 69    tstdata, trndata = alldata.splitWithProportion(0.25)
 70
 71    trndata._convertToOneOfMany()  # This is necessary, but I don't know why
 72    tstdata._convertToOneOfMany()  # http://stackoverflow.com/q/8154674/562769
 73
 74    print("Number of training patterns: %i" % len(trndata))
 75    print("Input and output dimensions: %i, %i" % (trndata.indim,
 76                                                   trndata.outdim))
 77    print("Hidden neurons: %i" % HIDDEN_NEURONS)
 78    print("First sample (input, target, class):")
 79    print(trndata['input'][0], trndata['target'][0], trndata['class'][0])
 80
 81    fnn = buildNetwork(trndata.indim, HIDDEN_NEURONS, trndata.outdim,
 82                       outclass=SoftmaxLayer)
 83
 84    trainer = BackpropTrainer(fnn, dataset=trndata, momentum=MOMENTUM,
 85                              verbose=True, weightdecay=WEIGHTDECAY)
 86
 87    # Visualization
 88    ticksX = arange(minX-1, maxX+1, 0.2)
 89    ticksY = arange(minY-1, maxY+1, 0.2)
 90    X, Y = meshgrid(ticksX, ticksY)
 91
 92    # need column vectors in dataset, not arrays
 93    griddata = ClassificationDataSet(INPUT_FEATURES, 1, nb_classes=CLASSES)
 94    for i in range(X.size):
 95        griddata.addSample([X.ravel()[i], Y.ravel()[i]], [0])
 96
 97    for i in range(20):
 98        trainer.trainEpochs(1)
 99        trnresult = percentError(trainer.testOnClassData(),
100                                 trndata['class'])
101        tstresult = percentError(trainer.testOnClassData(
102                                 dataset=tstdata), tstdata['class'])
103
104        print("epoch: %4d" % trainer.totalepochs,
105              "  train error: %5.2f%%" % trnresult,
106              "  test error: %5.2f%%" % tstresult)
107        out = fnn.activateOnDataset(griddata)
108        # the highest output activation gives the class
109        out = out.argmax(axis=1)
110        out = out.reshape(X.shape)
111
112        figure(1)  # always print on the same canvas
113        ioff()  # interactive graphics off
114        clf()   # clear the plot
115        for c in [0, 1, 2]:
116            here, _ = where(tstdata['class'] == c)
117            plot(tstdata['input'][here, 0], tstdata['input'][here, 1], 'o')
118        if out.max() != out.min():  # safety check against flat field
119            contourf(X, Y, out)   # plot the contour
120        ion()  # interactive graphics on
121        draw()  # update the plot
122
123    ioff()
124    show()
125
126if __name__ == '__main__':
127    from argparse import ArgumentParser, ArgumentDefaultsHelpFormatter
128
129    parser = ArgumentParser(formatter_class=ArgumentDefaultsHelpFormatter)
130
131    # Add more options if you like
132    parser.add_argument("-H", metavar="H", type=int, dest="hidden_neurons",
133                        default=5,
134                        help="number of neurons in the hidden layer")
135    parser.add_argument("-d", metavar="W", type=float, dest="weightdecay",
136                        default=0.01,
137                        help="weightdecay")
138    parser.add_argument("-m", metavar="M", type=float, dest="momentum",
139                        default=0.1,
140                        help="momentum")
141    args = parser.parse_args()
142
143    perceptron(args.hidden_neurons, args.weightdecay, args.momentum)
